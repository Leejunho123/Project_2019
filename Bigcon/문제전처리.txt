#ASFNT_DLY에 엑셀 함수를 이용하여 T1,T2를 추가한다.
#1. ASFNT_DLY.csv를 연다.
#2. 셀 서식 일반으로 지정한다.
#3. =hour(__)*60+MINUTE(__)을 이용해 T1변수를 만든다.(__에는 STT변수가 있는 첫번째 칸을 지정한다.)
#4. =DATE(A2+2016,B2,C2)*1440+T1 - DATE(2017,1,1)*1440-369을 이용해 T2변수를 만든다.
#5. ASFNT_DLY.csv로 저장한다.

#ASFNT_DLY.csv 불러오기
K<-read.csv(file.choose(),header=T)
K[,1]<-K[,1]-2016

#변인 제외하기
K<-K[,c(-7,-8,-10,-13,-14)]

#범주형 변수를 숫자 범주로 변환하기
K2<-matrix(0,nrow=nrow(K),ncol=ncol(K))
for(j in 4:7){
AC=levels(K[,j])
print(AC)
for(i in 1:length(AC)){
AD=which(K[,j]==AC[i])
K2[AD,j]=i
}
}

K2[,1]<-K[,1]
K2[,2]<-K[,2]
K2[,3]<-K[,3]
K2[,8]<-K[,8]
K2[,9]<-K[,9]

K2<-as.data.frame(K2)
write.csv(K2,"") #B.csv로 저장

#엑셀로 B.csv에서 2018년 9월 16일~2018년 9월 30일 기상데이터 추가하기
#2019년 9월 16일~2019년 9월 30일 기상데이터를 2018년 9월 16일~2018년 9월 30일 기상데이터로 사용
#기상데이터 결측치는 엑셀에서 필터를 활용하여 보간하였다.(기상데이터 추출 및 결측치 대체에 대한 방법은 결과보고서를 참조)
#데이터 추출과 결측치 보간이 된 기상데이터를 사용하였다.
#1. B.csv를 연다.
#2. B.csv에서 새로운 시트를 열어 하나의 공항(ex. ARP1)의 기상데이터를 붙여넣는다.
#3. 새로운 시트에 있는 기상데이터에서 2018년 평균기온 전체를 선택하여 INDEX를 평균기온으로 지정한다.
#4. 항공데이터에서 하나의 공항(ex. ARP1)에 필터를 넣는다.
#5. =IF(A__=3,INDEX(평균기온,B__+1,C__+1))을 이용하여 평균기온을 붙인다.(__에는 공항에 필터를 사용했을 때 나타나는 첫 번째 행의 번호를 넣는다.)
